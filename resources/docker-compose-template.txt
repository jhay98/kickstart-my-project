services:
  frontend:
    image: {docker-hub-repo}:{project-name}-frontend
    container_name: {project-name}-frontend
    build:
      context: ./frontend
      network: host
    ports:
      - 80:80
    env_file:
      - ./frontend/.env
    depends_on:
      - backend

  backend:
    image: {docker-hub-repo}:{project-name}-backend
    container_name: {project-name}-backend
    build:
      context: ./backend
      network: host
    env_file:
      - ./backend/.env
    depends_on:
      - db
    ports:
      - 8080:8080
    volumes:
      - ./logs:/home/logs

  db:
    image: postgres:alpine3.20
    build:
      network: host
    container_name: {project-name}postgres
    volumes:
      - postgres-data:/var/lib/postgresql/data
    env_file:
      - ./db/.env
    ports:
      - 5432:5432
